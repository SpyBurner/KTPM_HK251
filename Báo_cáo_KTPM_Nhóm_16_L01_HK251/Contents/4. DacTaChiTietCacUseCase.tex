\begin{usecase}{Cấu hình hệ thống} 
	\addrow{Nguồn}{\ref{us:A-US03}, \ref{nfr:01}} 
	\addrow{Use-case ID}{UC-SM-01} 
	\addrow{Tổng quan}{Admin chỉnh sửa các tham số hệ thống để tinh chỉnh hoạt động} 
	\addrow{Actor}{Admin} 
	\addcondition{Tiền điều kiện}{ 
		\item Hệ thống đang hoạt động ổn định. 
		\item Admin đã đăng nhập và có quyền. 
	} 
	\addcondition{Điều kiện kích hoạt}{\item Admin truy cập vào trang cấu hình hệ thống.} 
	\addmulrow{Các bước}{ 
		\item Hiển thị giao diện cấu hình tham số hệ thống. 
		\item Admin thay đổi các tham số cấu hình (lập lịch trích xuất, quản lý phạm vi trích xuất, mức hoạt động tối thiểu, hệ số đa dạng đề xuất). 
		\item Admin lưu thay đổi và xác nhận. 
		\item Hệ thống cập nhật và áp dụng cấu hình mới. 
	} 
	\addcondition{Hậu điều kiện}{ 
		\item Các tham số hệ thống được cập nhật thành công. 
		\item Dịch vụ liên quan áp dụng tham số mới. 
	} 
	\addexception{Xử lý ngoại lệ}{ 
		\item Không có thay đổi khi lưu, hệ thống hỏi xác nhận. 
		\item Lỗi lưu trữ, hiển thị thông báo lỗi. 
	} 
\end{usecase}

\begin{usecase}{Xem nhật ký hoạt động AI}
	\addrow{Nguồn}{\ref{us:A-US04}, \ref{nfr:18}}
	\addrow{Use-case ID}{UC-SM-02}
	\addrow{Tổng quan}{Admin xem nhật ký hoạt động (logs) của hệ thống AI để giám sát và phát hiện sự cố.}
	\addrow{Actor}{Admin}
	\addcondition{Tiền điều kiện}{
		\item Admin đã đăng nhập với quyền xem nhật ký.
	}
	\addcondition{Điều kiện kích hoạt}{
		\item Admin truy cập trang nhật ký hoạt động AI.
	}
	\addmulrow{Các bước}{
		\item Hệ thống hiển thị danh sách các bản ghi nhật ký AI, hỗ trợ lọc theo thời gian và mức độ nghiêm trọng. (Message, Warning, Error)
		\item Admin lựa chọn bản ghi hoặc khoảng thời gian cần xem.
		\item Hệ thống hiển thị chi tiết nhật ký theo yêu cầu.
	}
	\addcondition{Hậu điều kiện}{
		\item Admin có thể nắm bắt được thông tin hoạt động, cảnh báo hoặc lỗi của hệ thống AI.
	}
	\addexception{Xử lý ngoại lệ}{
		\item Nếu không có dữ liệu nhật ký trong khoảng thời gian yêu cầu, hệ thống hiển thị thông báo "Không có dữ liệu".
		\item Nếu xảy ra lỗi truy xuất dữ liệu nhật ký (ví dụ: mất kết nối), hệ thống thông báo lỗi và gợi ý tải lại trang.
	}
\end{usecase}

\begin{usecase}{Xem nhật ký kiểm toán (Audit log)}
	\addrow{Nguồn}{\ref{us:A-US04}}
	\addrow{Use-case ID}{UC-SM-03}
	\addrow{Tổng quan}{Admin xem nhật ký kiểm toán để giám sát các hành động hệ thống, đảm bảo an toàn và phát hiện hành vi bất thường.}
	\addrow{Actor}{Admin}
	\addcondition{Tiền điều kiện}{
		\item Admin đã đăng nhập với quyền xem audit log.
	}
	\addcondition{Điều kiện kích hoạt}{\item Admin truy cập trang nhật ký kiểm toán.}
	\addmulrow{Các bước}{
		\item Hệ thống hiển thị danh sách các sự kiện audit log, hỗ trợ lọc theo người dùng, loại sự kiện, thời gian.
		\item Admin chọn bộ lọc hoặc tìm kiếm cụ thể.
		\item Hệ thống hiển thị chi tiết các sự kiện phù hợp.
	}
	\addcondition{Hậu điều kiện}{
		\item Admin có thể kiểm tra và truy vết các thao tác hệ thống quan trọng.
	}
	\addexception{Xử lý ngoại lệ}{
		\item Nếu không có sự kiện nào phù hợp bộ lọc, hệ thống hiển thị thông báo "Không tìm thấy sự kiện".
		\item Nếu lỗi truy xuất dữ liệu audit log, hệ thống thông báo lỗi và yêu cầu tải lại.
	}
\end{usecase}

\begin{usecase}{Tạo hoặc Xóa tài khoản}
	\addrow{Nguồn}{\ref{us:A-US01}}
	\addrow{Use-case ID}{UC-AM-01}
	\addrow{Tổng quan}{Admin quản lý tài khoản người dùng: tạo mới hoặc xóa tài khoản khi cần thiết.}
	\addrow{Actor}{Admin}
	\addcondition{Tiền điều kiện}{
		\item Admin đã đăng nhập với quyền quản lý tài khoản.
	}
	\addcondition{Điều kiện kích hoạt}{\item Admin truy cập vào trang quản lý tài khoản.}
	\addmulrow{Các bước}{
		\item Admin chọn thao tác tạo tài khoản mới hoặc xóa tài khoản hiện có.
		\item Nếu tạo, Admin nhập các thông tin cần thiết (tên, email, quyền hạn,...).
		\item Hệ thống kiểm tra dữ liệu hợp lệ và tạo tài khoản.
		\item Nếu xóa, Admin chọn tài khoản cần xóa và xác nhận.
		\item Hệ thống xóa tài khoản và thông báo thành công.
	}
	\addcondition{Hậu điều kiện}{
		\item Tài khoản được tạo mới hoặc xóa khỏi hệ thống thành công.
	}
	\addexception{Xử lý ngoại lệ}{
		\item Nếu dữ liệu nhập không hợp lệ (ví dụ: email trùng, thiếu thông tin bắt buộc), hệ thống thông báo lỗi.
		\item Nếu tài khoản cần xóa đang liên kết với dữ liệu quan trọng (ví dụ: đang quản lý khóa học), hệ thống cảnh báo và yêu cầu xác nhận thêm hoặc từ chối xóa.
		\item Nếu lỗi hệ thống khi lưu hoặc xóa, thông báo lỗi và gợi ý thử lại.
	}
\end{usecase}

\begin{usecase}{Đặt lại mật khẩu (Reset password)}
	\addrow{Nguồn}{\ref{us:A-US01}}
	\addrow{Use-case ID}{UC-AM-02}
	\addrow{Tổng quan}{Admin có thể đặt lại mật khẩu cho tài khoản người dùng khi cần thiết.}
	\addrow{Actor}{Admin}
	\addcondition{Tiền điều kiện}{
		\item Admin đã đăng nhập .
		\item Admin có quyền đặt lại mật khẩu.
	}
	\addcondition{Điều kiện kích hoạt}{\item Admin truy cập trang quản lý tài khoản và chọn đặt lại mật khẩu.}
	\addmulrow{Các bước}{
		\item Admin chọn tài khoản người dùng cần đặt lại mật khẩu.
		\item Nhập mật khẩu mới hoặc hệ thống sinh mật khẩu ngẫu nhiên.
		\item Xác nhận thao tác đặt lại mật khẩu.
		\item Hệ thống cập nhật mật khẩu và thông báo thành công.
	}
	\addcondition{Hậu điều kiện}{
		\item Mật khẩu của tài khoản được thay đổi và có hiệu lực ngay lập tức.
	}
	\addexception{Xử lý ngoại lệ}{
		\item Nếu tài khoản không tồn tại hoặc không hợp lệ, hệ thống thông báo lỗi.
		\item Nếu lỗi hệ thống khi cập nhật, thông báo lỗi và yêu cầu thử lại.
	}
\end{usecase}

\begin{usecase}{Cấp và thu hồi quyền Trưởng khoa}
	\addrow{Nguồn}{\ref{us:A-US02}}
	\addrow{Use-case ID}{UC-PM-01}
	\addrow{Tổng quan}{Admin thực hiện cấp hoặc thu hồi quyền Trưởng khoa cho người dùng phù hợp để đảm bảo phân quyền đúng đối tượng.}
	\addrow{Actor}{Admin}
	\addcondition{Tiền điều kiện}{
		\item Admin đã đăng nhập.
		\item Admin có quyền quản lý quyền truy cập.
		\item Người dùng cần cấp/thu hồi quyền tồn tại trong hệ thống.
	}
	\addcondition{Điều kiện kích hoạt}{\item Admin truy cập trang quản lý quyền và chọn chức năng cấp hoặc thu hồi quyền Trưởng khoa.}
	\addmulrow{Các bước}{
		\item Admin tìm kiếm hoặc chọn người dùng cần thay đổi quyền.
		\item Admin chọn cấp quyền Trưởng khoa hoặc thu hồi quyền.
		\item Xác nhận thao tác cấp hoặc thu hồi.
		\item Hệ thống cập nhật quyền truy cập cho người dùng.
		\item Hệ thống hiển thị thông báo thực hiện thành công.
	}
	\addcondition{Hậu điều kiện}{
		\item Quyền Trưởng khoa được cập nhật chính xác theo yêu cầu.
	}
	\addexception{Xử lý ngoại lệ}{
		\item Nếu người dùng không tồn tại hoặc đã có quyền tương ứng, hệ thống thông báo lỗi và ngăn thao tác.
		\item Nếu lỗi hệ thống khi cập nhật quyền, hệ thống hiển thị lỗi và yêu cầu thử lại.
		\item Nếu Admin không có quyền thực hiện thao tác, hệ thống từ chối truy cập.
	}
\end{usecase}

\begin{usecase}{Phê duyệt yêu cầu quyền truy cập}

	\addrow{Actor}{Admin / Dean (Trưởng khoa)}
	\addcondition{Tiền điều kiện}{
		\item Admin hoặc Trưởng khoa đã đăng nhập với quyền phê duyệt.
		\item Có các yêu cầu quyền truy cập đang chờ xử lý.
	}
	\addcondition{Điều kiện kích hoạt}{\item Người có quyền truy cập vào trang phê duyệt yêu cầu quyền.}
	\addmulrow{Các bước}{
		\item Xem danh sách các yêu cầu cấp quyền đang chờ.
		\item Chọn từng yêu cầu để xem chi tiết.
		\item Phê duyệt hoặc từ chối yêu cầu.
		\item Hệ thống cập nhật trạng thái và thông báo kết quả cho người yêu cầu.
	}
	\addcondition{Hậu điều kiện}{
		\item Quyền truy cập được cấp hoặc yêu cầu bị từ chối chính xác.
		\item Người yêu cầu được thông báo kết quả kịp thời.
	}
	\addexception{Xử lý ngoại lệ}{
		\item Nếu yêu cầu không tồn tại hoặc đã được xử lý, hệ thống thông báo và không thực hiện thao tác.
		\item Nếu lỗi cập nhật quyền truy cập, hệ thống báo lỗi và cho phép thử lại.
		\item Nếu người phê duyệt không có quyền hợp lệ, hệ thống từ chối thao tác.
	}
\end{usecase}

\begin{usecase}{Thu hồi quyền truy cập}

	\addmulrow{Các bước}{
		\item Tìm hoặc chọn người dùng cần thu hồi quyền.
		\item Chọn quyền cần thu hồi.

		\item Hệ thống cập nhật và ghi nhận thay đổi.

	}
	\addcondition{Hậu điều kiện}{
		\item Quyền truy cập bị thu hồi có hiệu lực ngay lập tức.
	}
	\addexception{Xử lý ngoại lệ}{
		\item Nếu người dùng không tồn tại hoặc không có quyền được thu hồi, thông báo lỗi.
		\item Nếu lỗi hệ thống khi cập nhật, hiển thị thông báo và đề nghị thử lại.
		\item Nếu người thực hiện không đủ quyền, hệ thống từ chối thao tác.
	}
\end{usecase}

		\item Người dùng chọn loại quyền cần xin cấp.
		\item Nhập lý do hoặc mô tả nhu cầu sử dụng quyền.
		\item Gửi yêu cầu lên hệ thống.
		\item Hệ thống lưu yêu cầu và thông báo đã gửi thành công.
	}
	\addcondition{Hậu điều kiện}{
		\item Yêu cầu quyền được ghi nhận và chờ phê duyệt.

	\addexception{Xử lý ngoại lệ}{
